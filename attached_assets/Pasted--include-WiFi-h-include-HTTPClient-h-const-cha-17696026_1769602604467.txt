#include <WiFi.h>
#include <HTTPClient.h>

// ========== غيّر هذي القيم فقط ==========
const char* WIFI_SSID = "Camera";
const char* WIFI_PASS = "37570000";
// =========================================

const char* DEVICE_TOKEN = "GjbeSgWSIaQM8lizgcFdQUOnNSjstsS7";
const char* CHARGER_IP = "192.168.0.160";
const char* BARIQ_URL = "https://bariq-ev.replit.app/api/tesla-connector/vitals";
void setup() {
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("");
  Serial.println("================================");
  Serial.println("    Bariq - Tesla Charger");
  Serial.println("================================");
  Serial.println("");
  Serial.println("Connecting to WiFi...");
  
  WiFi.begin(WIFI_SSID, WIFI_PASS);
  
  int dots = 0;
  while (WiFi.status() != WL_CONNECTED && dots < 30) {
    delay(500);
    Serial.print(".");
    dots++;
  }
  
  Serial.println("");
  
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("WiFi Connected!");
    Serial.print("IP: ");
    Serial.println(WiFi.localIP());
  } else {
    Serial.println("WiFi Failed! Check SSID and password");
  }
  Serial.println("");
}

void loop() {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("WiFi disconnected! Reconnecting...");
    WiFi.reconnect();
    delay(5000);
    return;
  }
  
  Serial.println("--- Reading charger data ---");
  
  HTTPClient http;
  String chargerUrl = "http://" + String(CHARGER_IP) + "/api/1/vitals";
  
  http.begin(chargerUrl);
  http.setTimeout(10000);
  int httpCode = http.GET();
  
  if (httpCode == 200) {
    String vitals = http.getString();
    Serial.println("Charger data OK");
    http.end();
    
    Serial.println("--- Sending to Bariq ---");
    
    http.begin(BARIQ_URL);
    http.addHeader("Content-Type", "application/json");
    
    String payload = "{\"deviceToken\":\"";
    payload += DEVICE_TOKEN;
    payload += "\",\"vitals\":";
    payload += vitals;
    payload += "}";
    
    int postCode = http.POST(payload);
    
    if (postCode == 200) {
      String response = http.getString();
      Serial.println("Sent to Bariq OK!");
      Serial.println(response);
    } else {
      Serial.print("Bariq error: ");
      Serial.println(postCode);
    }
  } else {
    Serial.print("Charger error: ");
    Serial.println(httpCode);
  }
  
  http.end();
  Serial.println("");
  Serial.println("Waiting 30 seconds...");
  Serial.println("");
  
  delay(30000);
}
